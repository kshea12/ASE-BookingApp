import {
  Component,
  Input,
  Output,
  EventEmitter,
  ChangeDetectionStrategy
} from '@angular/core';

@Component({
  selector: 'ba-big-input-action',
  templateUrl: './big-input-action.component.html',
  styleUrls: ['./big-input-action.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class BigInputActionComponent {
  @Input() disabled = false;
  @Input() icon = '';
  @Input() label = '';
  @Input() color = '';

  @Output() action = new EventEmitter<void>();

  hasFocus = false;

  onClick() {
    this.action.emit();
  }
}

<button mat-raised-button color="color" [disabled]="disabled" (click)="onClick()">
  <mat-icon *ngIf="icon">{{icon}}</mat-icon>
  <span *ngIf="label">{{label}}</span>
</button>
